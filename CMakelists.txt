cmake_minimum_required(VERSION 3.18)
project(TradingSimulator LANGUAGES CXX)

# --- ADD THESE TWO LINES ---
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
# ---------------------------

find_package(cpr CONFIG REQUIRED)
find_package(nlohmann_json CONFIG REQUIRED)

# Group source files into a variable
set(TRADER_SOURCES
    data_extraction/api.cpp
    data_extraction/api_utils.cpp
    data_extraction/features_functions.cpp
    data_extraction/feature_utils.cpp
)

# Create the executable from the source file variable
add_executable(trader ${TRADER_SOURCES})

target_link_libraries(trader PRIVATE cpr::cpr nlohmann_json::nlohmann_json)
